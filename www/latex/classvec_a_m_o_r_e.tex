                N1.setId(10);
                N2.setId(20);
                N3.setId(30);

                Con1.setFromNeuron(&N1);

                MyvecCon.push_back(Con3);
                MyvecCon.show();
                MyvecCon.validate();

                vc = MyvecCon.getLdata();

                result.push_back(vc.at(0).getFromId());
                result.push_back(vc.at(1).getFromId());
                result.push_back(vc.at(2).getFromId());
